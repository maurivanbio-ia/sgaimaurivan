import { Router, Request, Response } from "express";

export const calendarioRouter = Router();

// Middleware opcional de autenticação
function requireAuth(req: Request, res: Response, next: Function) {
  const ok = Boolean((req as any).user || (req as any).session);
  if (!ok) return res.status(401).json({ message: "Não autenticado" });
  next();
}

calendarioRouter.get("/api/licencas", requireAuth, async (_req, res) => {
  try {
    const data = await getLicencas();
    res.setHeader("Content-Type", "application/json; charset=utf-8");
